Schritte

  • Renderer verwaltet ein <canvas> Element
  • Szene = PIXI.Container - Objekt
  • Objekte mittels .addChild hinzufügen
  • Rendern!
var renderer =
  PIXI.autoDetectRenderer(
    500, 768, {
      view: document.getElementById(
            'canvas-structure')
    });

var stage = new PIXI.Container();

var rect = new PIXI.Graphics()
  .beginFill(0xff0000)
  .drawRect(-100, -50, 200, 100)
  .endFill();
stage.addChild(rect);

rect.position.set(256, 384);

renderer.render(stage);
renderer.render(stage);
renderer.render(stage);
