Animationen

Eine Szene animieren

  • Szene verändern
  • Rendern
  • Neuen Animation-Frame anfordern
  • Beginne von vorne

Pixi gibt kein Framework für Animationen vor! 

var renderer = PIXI.autoDetectRenderer(
  500, 768, {
    view: document.getElementById(
          'canvas-animations')
  });

var stage = new PIXI.Container();

var rect = new PIXI.Graphics()
  .beginFill(0x0000ff)
  .drawRect(0, 0, 200, 100)
  .endFill();
rect.position.set(200, 300)
stage.addChild(rect);

function updateStage() {
  rect.rotation += 0.01;
}

function performAnimationLoop() {
  updateStage();
  renderer.render(stage);
  window.requestAnimationFrame(
    performAnimationLoop);
}

performAnimationLoop();

Ressourcen

Externe Ressourcen einbinden

  • Asynchron geladen mittels PIXI.loaders.Loader
  • Für Bilder und Spritesheets
  • Geladene Bilder automatisch in
    PIXI.utils.TextureCache
  • Neue Sprites mit PIXI.Sprite.fromFrame
var renderer = PIXI.autoDetectRenderer(
  500, 768, {
    view: document.getElementById(
          'canvas-resources')
  });

new PIXI.loaders.Loader()
  .add('img/pixi-logo.png')
  .add('img/cats.json')
  .on('complete', setupStage)
  .load();

function setupStage() {
  var stage = new PIXI.Container();
  var sprite = PIXI.Sprite
    .fromFrame('img/pixi-logo.png');

  stage.addChild(sprite);
  renderer.render(stage);
}

Interaktion

Den Interaktionsmanager nutzen

  • Objekte müssen .interactive=true sein
  • node-style Events (.on, .once...)
  • Eventnamen wie im Browser (mousemove, click...)
  • Auch Touch-Events werden unterstützt!
var renderer = PIXI.autoDetectRenderer(
  500, 768, {
    view: document.getElementById(
          'canvas-interactions')
  });
new PIXI.loaders.Loader()
  .add('img/cats.json')
  .on('complete', setupStage).load();

function setupStage() {
  var stage = new PIXI.Container();
  var sprite = PIXI.Sprite.fromFrame('0');
  stage.addChild(sprite);
  renderer.render(stage);
  sprite.interactive = true;

  sprite.on('mouseover', function () {
    var colorMatrix = new PIXI.filters
      .ColorMatrixFilter();
    colorMatrix.greyscale(0.5);
    sprite.filters = [colorMatrix];
    renderer.render(stage);
  });

  sprite.on('mouseout', function () {
    sprite.filters = null;
    renderer.render(stage);
  });
}
