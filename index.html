  • Container: Enthält andere Objekte

Objektkomposition

  • Alle Objekte sind auch Container
  • Objekte haben eine .position
  • Diese beschreibt die Position im Elternelement
  • Transformationen (.scale, .rotation)beeinflussen alle Kinder
var renderer = PIXI.autoDetectRenderer(
  500, 768, {
    view: document.getElementById(
          'canvas-coordinates')
  });

var stage = new PIXI.Container();

var circle = new PIXI.Graphics()
  .beginFill(0x00ff00)
  .drawCircle(0, 0, 100)
  .endFill();

var text = new PIXI.Text('Hallo');
circle.addChild(text);
stage.addChild(circle);

renderer.render(stage);

Animationen

Eine Szene animieren

  • Szene verändern
  • Rendern
  • Neuen Animation-Frame anfordern
  • Beginne von vorne

Pixi gibt kein Framework für Animationen vor! 

var renderer = PIXI.autoDetectRenderer(
  500, 768, {
    view: document.getElementById(
          'canvas-animations')
  });

var stage = new PIXI.Container();

var rect = new PIXI.Graphics()
  .beginFill(0x0000ff)
  .drawRect(0, 0, 200, 100)
  .endFill();
rect.position.set(200, 300)
stage.addChild(rect);

function updateStage() {
  rect.rotation += 0.01;
}

function performAnimationLoop() {
  updateStage();
  renderer.render(stage);
  window.requestAnimationFrame(
    performAnimationLoop);
}

performAnimationLoop();

Externe

Ressourcen

Externe Ressourcen einbinden

  • Asynchron geladen mittels PIXI.loaders.Loader
  • Für Bilder und Spritesheets
  • Einzelbilder automatisch in PIXI.utils.TextureCache
  • Neue Sprites mit PIXI.Sprite.fromFrame
var renderer = PIXI.autoDetectRenderer(
  500, 768, {
    view: document.getElementById(
          'canvas-resources')
  });

var loader = new PIXI.loaders.Loader()
  .add('img/tng.png')
  .add('img/cats.json')
  .on('complete', setupStage)
  .load();

function setupStage() {
  var stage = new PIXI.Container();
  var sprite = PIXI.Sprite
    .fromFrame('img/tng.png');
  stage.addChild(sprite);
  renderer.render(stage);
}
